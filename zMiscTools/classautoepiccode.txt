# Attributes
__file_name = None
__id = None

# Constructor
def __init__(self, file_name, id):
    self.setFileName(file_name)
    self.setId(id)

# Getters
def getFileName(self):
    return self.__file_name
def getId(self):
    return self.__id

# Setters
def setFileName(self, file_name):
    self.__file_name = file_name
def setId(self, id):
    self.__id = id
