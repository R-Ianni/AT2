# Attributes
__one = None
__two = None

# Constructor
def __init__(self, one, two):
    self.setOne(one)
    self.setTwo(two)

# Getters
def getOne(self):
    return self.__one
def getTwo(self):
    return self.__two

# Setters
def setOne(self, one):
    self.__one = one
def setTwo(self, two):
    self.__two = two
